    KNOWLEDGE_BASE_ARTICLES_USED                  INTEGER,
    PREVENTABLE_ISSUE                             BOOLEAN,
    PRIORITY_LEVEL                                VARCHAR(255),
    REASSIGNMENT_COUNT                            INTEGER,
    REOPENED_COUNT                                INTEGER,
    RESOLUTION_METHOD                             VARCHAR(255),
    ROOT_CAUSE_CATEGORY                           VARCHAR(255),
