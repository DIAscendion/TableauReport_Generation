-- STEP 1: CREATE WAREHOUSE
CREATE WAREHOUSE IF NOT EXISTS WH_ZOOM_DEV
    WAREHOUSE_SIZE = 'X-SMALL'
    AUTO_SUSPEND = 60
    AUTO_RESUME = TRUE
    COMMENT = 'Warehouse for Zoom Platform Analytics';

-- STEP 2: CREATE DATABASE
CREATE DATABASE IF NOT EXISTS DB_ZOOM_ANALYTICS
    COMMENT = 'Zoom Platform Analytics Database';

-- STEP 3: USE CONTEXT
USE WAREHOUSE WH_ZOOM_DEV;
USE DATABASE DB_ZOOM_ANALYTICS;

-- STEP 4: CREATE SCHEMA
CREATE SCHEMA IF NOT EXISTS DB_ZOOM_ANALYTICS.GOLD
